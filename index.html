                        <label for="note-title">Judul Catatan</label>
                        <input type="text" id="note-title" placeholder="Masukkan judul catatan" required>
                    </div>
                    <div class="form-group">
                        <label for="note-content">Isi Catatan</label>
                        <textarea id="note-content" placeholder="Tulis catatan Anda di sini" required></textarea>
                    </div>

                </form>
            </section>

Penjelasan Elemen Penting:

  • <form id="note-form">: Formulir yang akan menangani input pengguna. Atribut id penting agar bisa diakses oleh JavaScript.
  • <label for="...">: Label untuk meningkatkan aksesibilitas dan keterhubungan dengan input.
  • <input type="text" id="note-title">: Input teks satu baris untuk judul. id digunakan JavaScript untuk mengambil nilainya.
  • <textarea id="note-content">: Area teks multi-baris untuk isi catatan.
  • <button type="submit">: Tombol untuk mengirimkan formulir.
  • <div id="notes-container">: Wadah kosong yang akan diisi oleh catatan-catatan secara dinamis menggunakan JavaScript.

Notepad: Logika JavaScript

Menghidupkan aplikasi notepad dengan fungsi untuk menambah, menampilkan, dan menghapus catatan, serta menyimpannya di browser.

Kode Lengkap: script.js

Salin kode berikut dan simpan dalam file bernama script.js. Kode ini mengatur semua interaktivitas aplikasi.

// Tunggu hingga seluruh konten HTML dimuat
document.addEventListener('DOMContentLoaded', function() {
    // Mendapatkan elemen-elemen yang diperlukan dari DOM
    const noteForm = document.getElementById('note-form');
    const noteTitle = document.getElementById('note-title');
    const noteContent = document.getElementById('note-content');
    const notesContainer = document.getElementById('notes-container');
    
    // Mengambil catatan yang tersimpan di localStorage, atau buat array kosong jika tidak ada
    let notes = JSON.parse(localStorage.getItem('notes')) || [];
    
    // Menampilkan catatan yang ada saat halaman dimuat
    displayNotes();
    
    // Event listener untuk saat formulir disubmit
    noteForm.addEventListener('submit', function(e) {
        // Mencegah perilaku default form (reload halaman)
        e.preventDefault();
        
        // Membuat objek catatan baru
        const newNote = {
            id: Date.now(), // ID unik menggunakan timestamp
            title: noteTitle.value,
            content: noteContent.value,
            createdAt: new Date().toLocaleString()
        };
        
        // Menambahkan catatan baru ke array 'notes'
        notes.push(newNote);
        
        // Menyimpan array 'notes' ke localStorage
        saveNotes();
        
        // Memperbarui tampilan daftar catatan
        displayNotes();
        
        // Mengosongkan form setelah catatan disimpan
        noteForm.reset();
    });
    
    // Fungsi untuk menampilkan semua catatan ke dalam DOM
    function displayNotes() {
        // Mengosongkan container terlebih dahulu
        notesContainer.innerHTML = '';
        
        // Jika tidak ada catatan, tampilkan pesan
        if (notes.length === 0) {
            notesContainer.innerHTML = '<p class="empty-state">Belum ada catatan. Buat catatan baru!</p>';
            return;
        }
        
        // Loop melalui setiap catatan dan buat elemen HTML untuknya
        notes.forEach(note => {
            const noteElement = document.createElement('div');
            noteElement.classList.add('note-item');
            
            noteElement.innerHTML = `
                <h3 class="note-title">${note.title}</h3>
                <p class="note-content">${note.content}</p>
                <p class="note-date">Dibuat: ${note.createdAt}</p>
                <div class="note-actions">
                    <button class="delete-btn" data-id="${note.id}">Hapus</button>
                </div>
            `;
            
            // Menambahkan elemen catatan ke container
            notesContainer.appendChild(noteElement);
        });
        
        // Menambahkan event listener untuk setiap tombol hapus
        const deleteButtons = document.querySelectorAll('.delete-btn');
        deleteButtons.forEach(button => {
            button.addEventListener('click', function() {
                // Mendapatkan ID catatan dari atribut data-id
                const noteId = parseInt(this.getAttribute('data-id'));
                deleteNote(noteId);
            });
        });
    }
    
    // Fungsi untuk menghapus catatan berdasarkan ID
    function deleteNote(id) {
        // Filter array 'notes' untuk menghapus catatan dengan ID yang sesuai
        notes = notes.filter(note => note.id !== id);
        
        // Simpan perubahan ke localStorage
        saveNotes();
        
        // Perbarui tampilan
        displayNotes();
    }
    
    // Fungsi untuk menyimpan array 'notes' ke localStorage
    function saveNotes() {
        // localStorage hanya bisa menyimpan string, jadi kita ubah array menjadi string JSON
        localStorage.setItem('notes', JSON.stringify(notes));
    }
});

Penjelasan Logika Penting:

  • DOMContentLoaded: Memastikan kode JavaScript berjalan hanya setelah seluruh halaman HTML siap.
  • localStorage: Fitur browser untuk menyimpan data secara lokal. Data akan tetap ada bahkan setelah browser ditutup. JSON.stringify dan JSON.parse digunakan karena localStorage hanya bisa menyimpan teks.
  • addEventListener('submit', ...): Menangkap event saat tombol "Simpan" ditekan. e.preventDefault() mencegah halaman refresh.
  • document.createElement dan innerHTML: Cara JavaScript membuat dan menambahkan elemen HTML baru ke halaman secara dinamis.
  • data-id: Atribut khusus HTML untuk menyimpan data (dalam hal ini ID catatan) pada elemen, memudahkan akses dari JavaScript.

To-Do List: Logika JavaScript

Mengimplementasikan fungsi untuk menambah, menyelesaikan, menghapus tugas, dan menyimpannya menggunakan localStorage.

Kode Lengkap: script.js

Salin kode berikut dan simpan dalam file s>script.js. Kode ini mengatur semua logika aplikasi To-Do List.

document.addEventListener('DOMContentLoaded', () => {
    // Seleksi elemen-elemen DOM
    const taskForm = document.getElementById('task-form');
    const taskInput = document.getElementById('task-input');
    const taskUl = document.getElementById('task-ul');

    // Ambil tugas dari localStorage atau buat array kosong
    let tasks = JSON.parse(localStorage.getItem('tasks')) || [];

    // Fungsi untuk menyimpan tugas ke localStorage
    const saveTasks = () => {
        localStorage.setItem('tasks', JSON.stringify(tasks));
    };

    // Fungsi untuk merender daftar tugas
    const renderTasks = () => {
        taskUl.innerHTML = ''; // Kosongkan daftar terlebih dahulu
        tasks.forEach((task, index) => {
            const li = document.createElement('li');
            li.textContent = task.text;
            
            // Tambahkan kelas 'completed' jika tugas sudah selesai
            if (task.completed) {
                li.classList.add('completed');
            }

            // Buat tombol hapus
            const deleteButton = document.createElement('button');
            deleteButton.textContent = 'Hapus';
            deleteButton.addEventListener('click', () => deleteTask(index));

            // Tambahkan event listener untuk menandai tugas selesai
            li.addEventListener('click', () => toggleTaskComplete(index));

            // Masukkan tombol hapus ke dalam li
            li.appendChild(deleteButton);
            // Masukkan li ke dalam ul
            taskUl.appendChild(li);
        });
    };

    // Fungsi untuk menambah tugas baru
    const addTask = (e) => {
        e.preventDefault(); // Cegah form dari reload
        const text = taskInput.value.trim();
        if (text) {
            tasks.push({ text, completed: false });
            taskInput.value = ''; // Kosongkan input
            saveTasks();
            renderTasks();
        }
    };

    // Fungsi untuk menghapus tugas
    const deleteTask = (index) => {
        tasks.splice(index, 1); // Hapus tugas dari array
        saveTasks();
        renderTasks();
    };

    // Fungsi untuk toggle status selesai/tidak selesai
    const toggleTaskComplete = (index) => {
        tasks[index].completed = !tasks[index].completed;
        saveTasks();
        renderTasks();
    };

    // Event listener untuk submit form
    taskForm.addEventListener('submit', addTask);

    // Render tugas saat halaman dimuat pertama kali
    renderTasks();
});

Penjelasan Logika Penting:

  • tasks.push({ text, completed: false }): Menambahkan objek tugas baru ke array. Objek ini menyimpan teks dan status penyelesaian.
  • toggleTaskComplete(index): Fungsi ini mengubah nilai boolean completed dari false menjadi true atau sebaliknya saat item tugas diklik.
  • li.classList.add('completed'): JavaScript secara dinamis menambahkan kelas CSS ke elemen <li> untuk mengubah gayanya.
  • tasks.splice(index, 1): Metode array untuk menghapus satu elemen pada posisi index tertentu.

Tic-Tac-Toe: Logika JavaScript

Menulis logika inti permainan: mengatur giliran, menempatkan simbol, memeriksa kondisi menang/seri, dan me-reset game.

Kode Lengkap: script.js

Salin kode berikut dan simpan dalam file script.js. Ini adalah otak dari permainan Tic-Tac-Toe.

document.addEventListener('DOMContentLoaded', () => {
    const statusDisplay = document.querySelector('.status');
    const restartButton = document.querySelector('.restart-button');
    const cells = document.querySelectorAll('.cell');

    let gameActive = true;
    let currentPlayer = "X";
    let gameState = ["", "", "", "", "", "", "", "", ""];

    // Pesan yang akan ditampilkan
    const winningMessage = () => `Pemain ${currentPlayer} Menang!`;
    const drawMessage = () => `Permainan Seri!`;
    const currentPlayerTurn = () => `Giliranmu, ${currentPlayer}`;

    // Kondisi menang (8 kemungkinan)
    const winningConditions = [
        [0, 1, 2],
        [3, 4, 5],
        [6, 7, 8],
        [0, 3, 6],
        [1, 4, 7],
        [2, 5, 8],
        [0, 4, 8],
        [2, 4, 6]
    ];

    // Fungsi untuk menangani klik pada sel
    const handleCellPlayed = (clickedCell, clickedCellIndex) => {
        // Update state dan UI
        gameState[clickedCellIndex] = currentPlayer;
        clickedCell.innerHTML = currentPlayer;
        clickedCell.classList.add(currentPlayer.toLowerCase()); // Tambah kelas untuk styling
    };

    // Fungsi untuk mengganti pemain
    const handlePlayerChange = () => {
        currentPlayer = currentPlayer === "X" ? "O" : "X";
        statusDisplay.innerHTML = currentPlayerTurn();
    };

    // Fungsi untuk memeriksa hasil permainan
    const handleResultValidation = () => {
        let roundWon = false;
        for (let i = 0; i < winningConditions.length; i++) {
            const winCondition = winningConditions[i];
            const a = gameState[winCondition[0]];
            const b = gameState[winCondition[1]];
            const c = gameState[winCondition[2]];
            if (a === '' || b === '' || c === '') {
                continue; // Lanjut jika ada sel yang kosong
            }
            if (a === b && b === c) {
                roundWon = true;
                break; // Keluar dari loop jika ada yang menang
            }
        }

        if (roundWon) {
            statusDisplay.innerHTML = winningMessage();
            gameActive = false;
            return;
        }

        // Cek apakah permainan seri
        const roundDraw = !gameState.includes("");
        if (roundDraw) {
            statusDisplay.innerHTML = drawMessage();
            gameActive = false;
            return;
        }

        // Jika belum ada yang menang atau seri, ganti pemain
        handlePlayerChange();
    };

    // Fungsi untuk menangani klik pada sel
    const handleCellClick = (event) => {
        const clickedCell = event.target;
        const clickedCellIndex = parseInt(clickedCell.getAttribute('data-cell-index'));

        // Cegah klik jika sel sudah terisi atau game sudah berakhir
        if (gameState[clickedCellIndex] !== "" || !gameActive) {
            return;
        }

        handleCellPlayed(clickedCell, clickedCellIndex);
        handleResultValidation();
    };

    // Fungsi untuk me-restart permainan
    const handleRestartGame = () => {
        gameActive = true;
        currentPlayer = "X";
        gameState = ["", "", "", "", "", "", "", "", ""];
        statusDisplay.innerHTML = currentPlayerTurn();
        cells.forEach(cell => {
            cell.innerHTML = "";
            cell.classList.remove('x', 'o'); // Hapus kelas styling
        });
    };

    // Tambahkan event listener ke setiap sel dan tombol restart
    cells.forEach(cell => cell.addEventListener('click', handleCellClick));
    restartButton.addEventListener('click', handleRestartGame);
});

Penjelasan Logika Penting:

  • gameState: Array yang menyimpan status setiap sel (kosong, "X", atau "O"). Ini adalah "sumber kebenaran" tunggal untuk kondisi papan.
  • winningConditions: Array 2D yang berisi semua indeks sel yang membentuk garis kemenangan (3 baris, 3 kolom, 2 diagonal).
  • handleResultValidation(): Fungsi krusial yang memeriksa apakah ada pemain yang menang dengan mencocokkan gameState dengan winningConditions, atau apakah permainan seri.
  • gameActive: Variabel boolean untuk mencegah interaksi (seperti mengklik sel) setelah permainan berakhir.

JavaScript Asinkron: `async/await`

Gunakan sintaks `async/await` untuk menulis kode asinkron yang terlihat dan berperilaku seperti kode sinkron, membuatnya lebih mudah dibaca dan dikelola, terutama saat mengambil data dari API.

async function fetchUserData() {
  try {
    const response = await fetch('https://api.github.com/users/google');
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    const data = await response.json();
    console.log(data.name); // Google
  } catch (error) {
    console.error('Gagal mengambil data:', error);
  }
}

Hooks: `useState` & `useEffect`

`useState` untuk mengelola state lokal. `useEffect` untuk menangani efek samping seperti fetch data atau berlangganan event. Keduanya adalah fondasi dari komponen fungsional modern.

import React, { useState, useEffect } from 'react';

function UserProfile({ userId }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    fetch(`https://api.example.com/users/${userId}`)
      .then(res => res.json())
      .then(data => setUser(data));
  }, [userId]); // Jalankan lagi jika userId berubah

  if (!user) return <p>Loading...</p>;
  return <h1>{user.name}</h1>;
}

class Mobil:
